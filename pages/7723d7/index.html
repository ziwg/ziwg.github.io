docker-compose本地搭建常用服务

# docker-compose常用命令
#启动文件内所有服务 文件名为 docker-compose.yml 时可省略
docker-compose -f docker-compose.yml up
#后台启动单个服务
docker-compose up -d 服务名
#停止服务
docker-compose stop 服务名
#启动已存在的服务
docker-compose start 服务名
#删除服务
docker-compose rm 服务名
#进入容器
docker-compose exec 服务名 sh(bash)

# docker-compose.yml
version: '3'
services:
  redis: 
    image: redis:5.0.8
    container_name: redis
    restart: always
    volumes:
      - ./data/redis:/data
      - ./config/redis:/usr/local/etc/redis
    ports:
      - 6379:6379
    command: redis-server /usr/local/etc/redis/redis.conf
  mysql:
    image: mysql:5.7
    container_name: mysql
    restart: always
    ports:
      - 3306:3306
    volumes:
      - ./data/mysql:/var/lib/mysql
    environment:
      TZ: Asia/Shanghai
      MYSQL_ROOT_PASSWORD: 123456
